        - index.php
        - -S
        - 0.0.0.0:8080

        - php
      image: php:alpine
      name: phpinfo-container
      ports:
        - containerPort: 8080
          protocol: TCP

        - mountPath: /app/index.php
          name: phpinfo-volume
          readOnly: true

    runAsUser: 65534
